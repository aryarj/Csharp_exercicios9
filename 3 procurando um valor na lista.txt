using System;
using System.Collections.Generic;

public class Program
{
	public static void Main()
	{
		/* Console.Clear();*/
		List<int> lista=new List<int>();
         Console.WriteLine("Informa se um número digitado se encontra ou não na lista");
		 Console.WriteLine("de 10 números aleatórios de 1 a 10");
        
		 int qtde=10;
         Random rn = new Random();
		 int a;
		 int n;
		 int countn=0;
		 		
		 Console.WriteLine("Digite o valor a ser encontrado");
		 n=Convert.ToInt32(Console.ReadLine());
		
		//fabricando uma lista com 10 valores aleatórios de 1 a 10
		Console.WriteLine("lista com 10 números aleatórios de 1 a 10");
		 for(int i=0; i<qtde ; i++)
		 {
			 a=rn.Next(1,10);

			 Console.WriteLine("Número na lista "+a);
		 }
        
		 //contando os numeros iguais a n na lista
          foreach(int aux in lista)
         {			  
			  if(aux==n)
			  {countn++;}
	     }

		{Console.WriteLine("Foram encontrados "+countn+" números "+n+" na lista");}

		{Console.WriteLine("NÃO foram encontrados números "+n+" na lista");}
						  
	}
